names = [" Alice ", "bob", " CHARLIE "]
cleaned_names = []
#clean each name
for name in names:
    name = name.strip()  #removes extra spaces
    name = name.lower()  #converts to lowercase
    cleaned_names.append(name)
print("Cleaned Names:", cleaned_names)
